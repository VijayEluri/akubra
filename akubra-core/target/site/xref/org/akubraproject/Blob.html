22  package org.akubraproject;
23  
24  import java.io.IOException;
25  import java.io.InputStream;
26  import java.io.OutputStream;
27  
28  import java.net.URI;
29  import java.util.Map;
30  
31  /**
32   * Interface to abstract the idea of a blob in the blob store.
33   *
34   * @author Chris Wilper
35   * @author Pradeep Krishnan
36   * @author Ronald Tschal√§r
37   */
38  public interface Blob {
39    /**
40     * Gets the connection that provided this blob.
41     *
42     * @return the blob store connection that created this blob
43     */
44    BlobStoreConnection getConnection();
45  
46    /**
47     * Gets the id of the blob. Blob ids are URIs as defined by RFC 3986, and therefore will
48     * always have a scheme.
49     *
50     * @return the non-null immutable blob id
51     */
52    URI getId();
53  
54    /**
55     * Gets the canonical id of the blob, if known. The definition of what is canonical is
56     * up to the individual store implementation. When available, the canonical id should be
57     * used in caches and other indexes where duplication is undesirable.
58     *
59     * @return the canonical id, if known; null otherwise.
60     * @throws IOException if an error occurred
61     */
62    URI getCanonicalId() throws IOException;
63  
64    /**
65     * Opens a new InputStream for reading the content.
66     *
67     * @return the input stream.
68     * @throws MissingBlobException if the blob does not {@link #exists exist}.
69     * @throws IOException if the stream cannot be opened for any other reason.
70     */
71    InputStream openInputStream() throws IOException, MissingBlobException;
72  
73    /**
74     * Opens a new OutputStream for writing the content. If the blob does not exist it is created
75     * with the data written to the stream as the content; else if <var>overwrite</var> is true the
76     * existing content is replaced with the new data.
77     *
78     * @param estimatedSize the estimated size of the data if known (or -1 if unknown).
79     *                      This can allow for the implementation to make better decisions
80     *                      on buffering or reserving space.
81     * @param overwrite     if false and this blob already exists then do not overwrite the contents
82     *                      but throw an exception instead
83     * @return the output stream.
84     * @throws DuplicateBlobException if this blob exists and <var>overwrite</var> is false
85     * @throws IOException if the stream cannot be opened for any other reason.
86     */
87    OutputStream openOutputStream(long estimatedSize, boolean overwrite)
88        throws IOException, DuplicateBlobException;
89  
90    /**
91     * Gets the size of the blob, in bytes.
92     *
93     * @return the size in bytes, or -1 if unknown
94     * @throws MissingBlobException if the blob does not {@link #exists exist}.
95     * @throws IOException if an error occurred during
96     */
97    long getSize() throws IOException, MissingBlobException;
98  
99    /**
100    * Tests if a blob with this id exists in this blob-store.
101    *
102    * @return true if the blob denoted by this id exists; false otherwise.
103    *
104    * @throws IOException if an error occurred during existence check
105    * @see #openOutputStream
106    * @see #delete
107    */
108   boolean exists() throws IOException;
109 
110   /**
111    * Removes this blob from the store. This operation is idempotent and does
112    * not throw an exception if the blob does not {@link #exists exist}.
113    *
114    * @throws IOException if the blob cannot be deleted for any reason
115    */
116   void delete() throws IOException;
117 
118   /**
119    * Move this blob under the new id. Before the move, this blob must exist and the destination
120    * blob must not. After the move, this blob will not exist but the destination blob will.
121    *
122    * @param blobId the blob id; may be null if the store supports id-generation
123    * @param hints A set of hints to allow the implementation to optimize the operation (can be
124    *              null)
125    *
126    * @return the resulting Blob from the move
127    *
128    * @throws UnsupportedIdException if blobId is not in a recognized/usable pattern by this store
129    * @throws UnsupportedOperationException if <var>blobId</var> is null and this store is not
130    *                                       capable of generating ids.
131    * @throws MissingBlobException if this blob does not exist
132    * @throws DuplicateBlobException if a blob with <var>blobId</var> already exists
133    * @throws IOException if an error occurs while attempting the operation
134    */
135   Blob moveTo(URI blobId, Map<String, String> hints)
136       throws DuplicateBlobException, IOException, MissingBlobException, NullPointerException,
137              IllegalArgumentException;
138 }
