22  package org.akubraproject;
23  
24  import java.io.IOException;
25  import java.io.InputStream;
26  
27  import java.net.URI;
28  
29  import java.util.Iterator;
30  import java.util.Map;

32  /**
33   * Interface to abstract the idea of a connection to a transaction based blob store
34   *
35   * @author Chris Wilper
36   * @author Pradeep Krishnan
37   * @author Ronald Tschal√§r
38   */
39  public interface BlobStoreConnection {
40    /**
41     * Gets the blob store associated with this connection.
42     *
43     * @return the blob store.
44     */
45    BlobStore getBlobStore();
46  
47    /**
48     * Gets the blob with the given id.
49     *
50     * @param blobId the blob id; may be null if the store supports id-generation
51     * @param hints A set of hints to allow the implementation to optimize the operation (can be
52     *              null)
53     *
54     * @return the blob. Cannot be null and must have the passed in blobId if blobId is not null. If
55     *         the passed in blobId is null, a new and unique (for the store) id must be generated if
56     *         the store is capable of generating ids; the thus returned blob may exist, in which
57     *         case it will have an empty content. In either case there is no requirement that the
58     *         returned blob must {@link Blob#exists exist}. However there is a requirement that the
59     *         {@link Blob#getConnection getConnection()} method of the returned Blob must return
60     *         this connection object.
61     *
62     * @throws IOException for IO errors
63     * @throws UnsupportedIdException if blobId is not in a recognized/usable pattern by this store
64     * @throws UnsupportedOperationException if <var>blobId</var> is null and this store is not
65     *                                       capable of generating ids.
66     */
67    Blob getBlob(URI blobId, Map<String, String> hints)
68          throws IOException, UnsupportedIdException, UnsupportedOperationException;
69  
70    /**
71     * Creates a blob with the given content. For Content Addressible Storage (CAS) systems,
72     * this is the only way to create a Blob; for all other stores there is also {@link #getBlob(URI,
73     * Map) getBlob}.{@link Blob#openOutputStream openOutputStream}.
74     *
75     * @param content the contents of the blob
76     * @param estimatedSize the estimated size of the data if known (or -1 if unknown).
77     *                      This can allow for the implementation to make better decisions
78     *                      on buffering or reserving space.
79     * @param hints A set of hints to allow the implementation to optimize the operation (can be
80     *              null)
81     *
82     * @return the blob. Cannot be null and must have a generated id. The {@link Blob#getConnection
83     *                   getConnection()} method must return this connection object.
84     *
85     * @throws IOException for IO errors
86     * @throws UnsupportedOperationException if this store cannot generate new id and create a new
87     *                                       blob
88     */
89    Blob getBlob(InputStream content, long estimatedSize, Map<String, String> hints)
90          throws IOException, UnsupportedOperationException;
91  
92    /**
93     * Gets an iterator over the ids of all blobs in this store.
94     *
95     * @param filterPrefix If not null, the list will be limited to those blob-ids beginning with
96     *                     this prefix.
97     *
98     * @return The iterator over the blob-ids.
99     * @throws IOException if an error occurred getting the list of blob ids
100    */
101   Iterator<URI> listBlobIds(String filterPrefix) throws IOException;
102 
103   /**
104    * Flush all blobs associated with this connection and fsync. After this method completes all
105    * data should be comitted to stable storage (for stores that are backed by stable storage).
106    * However, whether this is really the case will depend on the storage used by the store (e.g.
107    * for a local filesystem based store it'll depend on the OS config and hardware, i.e. whether
108    * the disk's write cache is flushed or not).
109    *
110    * <p>This may or may not flush any open output-streams. For reliable results ensure that all
111    * output streams are closed before calling sync.
112    *
113    * @throws IOException if any error occurred trying to flush and sync
114    * @throws UnsupportedOperationException if the store is not capable of syncing
115    */
116   void sync() throws IOException, UnsupportedOperationException;
117 
118   /**
119    * Close the connection to the blob store. After this, all Blob and BlobStoreConnection
120    * operations except for {@link #getBlobStore getBlobStore}, {@link #isClosed isClosed},
121    * {@link #close close}, {@link Blob#getId Blob.getId}, and {@link Blob#getConnection
122    * Blob.getConnection} will throw an {@link java.lang.IllegalStateException
123    * IllegalStateException}.
124    *
125    * <p>This method is idempotent and may be called multiple times.
126    */
127   void close();
128 
129   /**
130    * Tests if the connection to the blob store is closed.
131    *
132    * @return true if the connection is closed.
133    * @see #close
134    */
135   boolean isClosed();
136 }
