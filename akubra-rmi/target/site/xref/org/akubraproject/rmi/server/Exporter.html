42  /**
43   * RMI object exporter. Encapsulates the configuration needed to export an RMI object. It also
44   * holds a reference to the exported objects so that exported objects stays alive until
45   * un-exported.
46   *
47   * @author Pradeep Krishnan
48   */
49  public class Exporter {
50    private static final Logger  log              = LoggerFactory.getLogger(Exporter.class);
51  
52    /**
53     * The number of milliseconds to wait before re-attempting an unexport. Note that
54     * there is a bit of a back-off mechanism - each subsequent retry will use a multiple
55     * of this delay.
56     */
57    public static final long RETRY_DELAY = 10;
58    private final int                          port;
59    private final RMIClientSocketFactory       csf;
60    private final RMIServerSocketFactory       ssf;
61    private final ScheduledExecutorService     executor;
62    private final Set<Exportable>              exportedObjects;
63  
64    /**
65     * Creates a new Exporter object.
66     *
67     * @param port the port number on which the remote object receives calls (if port is zero, an
68     *          anonymous port is chosen)
69     */
70    public Exporter(int port) {
71      this(port, null, null);
72    }
73  
74    /**
75     * Creates a new Exporter object.
76     *
77     * @param port the port number on which the remote object receives calls (if port is zero, an
78     *          anonymous port is chosen)
79     * @param csf the client-side socket factory for making calls to the remote object
80     * @param ssf the server-side socket factory for receiving remote calls
81     */
82    public Exporter(int port, RMIClientSocketFactory csf, RMIServerSocketFactory ssf) {
83      this.port         = port;
84      this.csf          = csf;
85      this.ssf          = ssf;
86      this.executor     = createExecutor();
87      exportedObjects   = Collections.synchronizedSet(new HashSet<Exportable>());
88    }
89  
90    private static ScheduledExecutorService createExecutor() {
91      return Executors.newSingleThreadScheduledExecutor(new ThreadFactory() {
92            public Thread newThread(Runnable r) {
93              Thread t = new Thread(r, "akubra-rmi-unexporter");
94              t.setPriority(Thread.MIN_PRIORITY);
95              t.setDaemon(true);
96  
97              return t;
98            }
99          });
100   }
101 
102   /**
103    * Gets the port where the exported objects listen.
104    *
105    * @return the port
106    */
107   public int getPort() {
108     return port;
109   }
110 
111   /**
112    * Exports the remote object to make it available to receive incoming calls using this
113    * configuration. Note that this exporter holds a reference to the exported object (not the
114    * stub, but the Exportable itself) so that the exported object remains in scope and does not
115    * get garbage collected. (This is something that the RMI spec claims to provide, but in reality
116    * does not. The sun.rmi.transport.ObjectTable only keeps weak references.) So this means, all
117    * exported objects using this exporter must be {@link #unexportObject un-exported} using this
118    * so that the reference held here is cleared. Failure to do so will result in memory leaks.
119    *
120    * @param object the remote object to be exported
121    *
122    * @return remote object stub
123    *
124    * @throws RemoteException if export fails
125    */
126   public Remote exportObject(Exportable object) throws RemoteException {
127     Remote stub =
128       (csf == null) ? UnicastRemoteObject.exportObject(object, port)
129       : UnicastRemoteObject.exportObject(object, port, csf, ssf);
130     exportedObjects.add(object);
131 
132     return stub;
133   }
134 
135   /**
136    * Removes the remote object from the RMI runtime. If successful, the object can no longer
137    * accept incoming RMI calls. If the force parameter is true, the object is forcibly unexported
138    * even if there are pending calls to the remote object or the remote object still has calls in
139    * progress. If the force parameter is false, the object is only unexported when there are no
140    * pending or in progress calls to the object. If there are pending calls, the unexport is
141    * re-scheduled for later.
142    *
143    * @param object the remote object to be unexported
144    * @param force if true, unexports the object even if there are pending or in-progress calls; if
145    *        false, only unexports the object when there are no pending or in-progress calls
146    *
147    * @throws NoSuchObjectException if the object is not exported
148    */
149   public void unexportObject(Exportable object, boolean force)
150                       throws NoSuchObjectException {
151     exportedObjects.remove(object);
152     unexportObject(object, force, 0);
153   }
154 
155   private void unexportObject(Remote object, boolean force, int trial)
156                        throws NoSuchObjectException {
157     if (!UnicastRemoteObject.unexportObject(object, force))
158       scheduleRetry(object, force, trial);
159   }
160 
161   private void scheduleRetry(final Remote object, final boolean force, final int trial) {
162     if (log.isDebugEnabled())
163       log.debug("Scheduling retry #" + trial + " of unexport for " + object);
164 
165     Runnable job =
166       new Runnable() {
167         public void run() {
168           try {
169             if (log.isDebugEnabled())
170               log.debug("Retrying unexport for " + object);
171 
172             unexportObject(object, force, trial + 1);
173           } catch (NoSuchObjectException e) {
174             if (log.isDebugEnabled())
175               log.debug(object + " is already unexported", e);
176           }
177         }
178       };
179 
180     executor.schedule(job, RETRY_DELAY * trial, TimeUnit.MILLISECONDS);
181   }
182 
183   /**
184    * Gets the set of all exported objects.
185    *
186    * @return a copy of the exported objects.
187    */
188   public Set<Exportable> getExportedObjects() {
189     synchronized (exportedObjects) {
190       return new HashSet<Exportable>(exportedObjects);
191     }
192   }
